  \begin{alignedat}{1}
  0 &=
    (m_a +m_b)l_a(c_{a-b}l_a\ddot{\theta_a}

     -m_bs_{a-b}c_{a-b}l_al_b\dot{\theta_b}^2 \\

     +(m_a +m_b)(s_b -c_{a-b}s_a)l_ag
  \end{alignedat}
  $$

  $$
  \begin{alignedat}{1}
  m_a& +m_b -m_bc_{a-b}^2 \\
  &= m_a +m_b(1 - c_{a-b}^2) \\
  &= m_a +m_bs_{a-b}^2 \\
  \end{alignedat}

  \begin{alignedat}{1}
  s_b& -c_{a-b}s_a \\
  &= s_ac_ac_b+s_a^2s_b -s_b \\
  &= s_{a-b}c_a +c_a^2s_b +s_a^2s_b -s_b \\
  &= s_{a-b}c_a +(c_a^2 +s_a^2 -1)s_b \\
  &= s_{a-b}c_a \\
  \end{alignedat}
  $$

     -m_bs_{a-b}c_{a-b}l_al_b\dot{\theta_b}^2
     -(m_a +m_b)s_{a-b}c_al_ag
  \end{alignedat}
  $$
